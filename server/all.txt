db schema

employees
CREATE TABLE employees (id SERIAL PRIMARY KEY, name VARCHAR(100) NOT NULL, email VARCHAR(255) UNIQUE NOT NULL, role VARCHAR(20) CHECK (role IN ('EMPLOYEE', 'MANAGER', 'TEAMLEAD')) DEFAULT 'EMPLOYEE', team VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

recognitions
CREATE TABLE recognitions (id SERIAL PRIMARY KEY, sender_id INT REFERENCES employees(id), recipient_id INT REFERENCES employees(id), message TEXT, emoji VARCHAR(10), visibility VARCHAR(10) CHECK (visibility IN ('PUBLIC', 'PRIVATE', 'ANONYMOUS')) DEFAULT 'PUBLIC', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
